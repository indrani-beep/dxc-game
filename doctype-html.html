            // Functions

            function startGame() {

                gameState.currentScenario = 0;

                gameState.score = 0;

                gameState.scenarioResults = [];

                updateScore();

                updateScenario();

                startTimer();

                createAnimatedAvatar();

               

                // Generate random position for the avatar

                animateAvatar();

               

                // Create hint popup with delay

                setTimeout(showHint, 3000);

            }

           

            function updateScenario() {

                const scenario = scenarios[gameState.currentScenario];

               

                // Update scenario information

                scenarioTitle.textContent = scenario.title;

                scenarioDescription.textContent = scenario.description;

                scenarioNumber.textContent = gameState.currentScenario + 1;

               

                // Create option elements

                optionsContainer.innerHTML = '';

                scenario.options.forEach((option, index) => {

                    const optionElement = document.createElement('div');

                    optionElement.className = 'scenario-option slide-in-right';

                    optionElement.style.animationDelay = `${0.1 * index}s`;

                    optionElement.textContent = option.text;

                    optionElement.addEventListener('click', () => selectOption(index));

                    optionsContainer.appendChild(optionElement);

                });

               

                // Update hint

                hintText.textContent = scenario.hint;

                hideElement(hintContainer);

               

                // Update background

                sceneBackground.className = 'scene-background';

                sceneBackground.classList.add(scenario.backgroundClass);

            }

           

            function selectOption(index) {

                // Clear timer

                clearInterval(gameState.timerInterval);

               

                const scenario = scenarios[gameState.currentScenario];

                const selectedOption = scenario.options[index];

                const isCorrect = selectedOption.correct;

               

                // Update score

                if (isCorrect) {

                    gameState.score += 10;

                    playSound(correctSound);

                } else {

                    gameState.score = Math.max(0, gameState.score - 5);

                    playSound(incorrectSound);

                }

                updateScore();

               

                // Store result

                gameState.scenarioResults.push({

                    scenario: scenario.title,

                    isCorrect: isCorrect

                });

               

                // Show feedback

                feedbackTitle.textContent = isCorrect ? "Great choice!" : "Consider this approach...";

                feedbackText.textContent = selectedOption.feedback;

                feedbackContainer.className = isCorrect ?

                    'bg-green-50 text-green-800 border-l-4 border-green-400 p-4 rounded-lg fade-in' :

                    'bg-orange-50 text-orange-800 border-l-4 border-orange-400 p-4 rounded-lg fade-in';

               

                // Highlight selected option

                const optionElements = optionsContainer.querySelectorAll('.scenario-option');

                optionElements.forEach((element, i) => {

                    if (i === index) {

                        element.classList.add(isCorrect ? 'correct' : 'incorrect');

                    }

                });

               

                // Show feedback after a short delay

                setTimeout(() => {

                    hideElement(scenarioContainer);

                    showElement(feedbackContainer);

                }, 1000);

               

                // Update avatar to react to choice

                const avatar = document.querySelector('.animated-avatar');

                if (avatar) {

                    avatar.classList.remove('floating');

                    avatar.classList.add(isCorrect ? 'bounce' : 'shake');

                    setTimeout(() => {

                        avatar.classList.remove('bounce', 'shake');

                        avatar.classList.add('floating');

                    }, 1000);

                }

            }

           

            function startTimer() {

                gameState.timeLeft = 10;

                updateTimerDisplay();

               

                clearInterval(gameState.timerInterval);

                gameState.timerInterval = setInterval(() => {

                    gameState.timeLeft--;

                    updateTimerDisplay();

                   

                    if (gameState.timeLeft <= 0) {

                        clearInterval(gameState.timerInterval);

                        // Auto-select the worst option if time runs out

                        const worstOptionIndex = scenarios[gameState.currentScenario].options.findIndex(option => !option.correct);

                        selectOption(worstOptionIndex >= 0 ? worstOptionIndex : 0);

                    }

                }, 1000);

            }

           

            function updateTimerDisplay() {

                timerDisplay.textContent = gameState.timeLeft;

                const percentage = (gameState.timeLeft / 10) * 100;

                timerCircle.style.background = `conic-gradient(var(--dxc-purple) 0% ${percentage}%, transparent ${percentage}% 100%)`;

               

                if (gameState.timeLeft <= 3) {

                    timerDisplay.classList.add('text-red-500');

                    timerCircle.classList.add('pulse');

                } else {

                    timerDisplay.classList.remove('text-red-500');

                    timerCircle.classList.remove('pulse');

                }

            }

           

            function showResults() {

                hideScreen(gameScreen);

                showScreen(resultsScreen);

               

                // Display final score

                finalScoreDisplay.textContent = gameState.score;

               

                // Calculate assessment

                const correctAnswers = gameState.scenarioResults.filter(result => result.isCorrect).length;

                const totalScenarios = scenarios.length;

                const percentage = (correctAnswers / totalScenarios) * 100;

               

                // Set assessment text based on percentage

                let assessment;

                if (percentage >= 80) {

                    assessment = "Excellent alignment with DXC values! You demonstrate a strong understanding of our collaborative and client-focused approach.";

                    createConfetti(100);

                } else if (percentage >= 60) {

                    assessment = "Good alignment with DXC values. You show potential in understanding our culture with some areas for growth.";

                    createConfetti(50);

                } else {

                    assessment = "There are opportunities to better align with DXC's values and approach. We encourage learning more about our collaborative culture.";

                    createConfetti(20);

                }

                assessmentText.textContent = assessment;

               

                // Generate strengths and growth areas

                updateStrengthsAndGrowthAreas();

               

                // Create leaderboard

                updateLeaderboard();

            }

           

            function updateStrengthsAndGrowthAreas() {

                strengthsList.innerHTML = '';

                growthList.innerHTML = '';

               

                // Map each scenario to a skill area

                const skillAreas = [

                    "Conflict Resolution",

                    "Virtual Collaboration",

                    "Work-Life Balance",

                    "Empathy & Support",

                    "Client Relations"

                ];

               

                gameState.scenarioResults.forEach((result, index) => {

                    const listItem = document.createElement('li');

                    listItem.className = 'flex items-start';

                   

                    const icon = document.createElement('span');

                    icon.className = result.isCorrect ? 'text-green-600 mr-2' : 'text-orange-600 mr-2';

                    icon.innerHTML = result.isCorrect ? '<i class="fas fa-check-circle"></i>' : '<i class="fas fa-exclamation-circle"></i>';

                   

                    const text = document.createElement('span');

                    text.textContent = skillAreas[index];

                   

                    listItem.appendChild(icon);

                    listItem.appendChild(text);

                   

                    if (result.isCorrect) {

                        strengthsList.appendChild(listItem);

                    } else {

                        growthList.appendChild(listItem);

                    }

                });

               

                // If all correct or all incorrect, add a message

                if (strengthsList.children.length === 0) {

                    const listItem = document.createElement('li');

                    listItem.textContent = "We'll identify strengths as you develop your skills.";

                    strengthsList.appendChild(listItem);

                }

               

                if (growthList.children.length === 0) {

                    const listItem = document.createElement('li');

                    listItem.textContent = "Excellent! No specific growth areas identified.";

                    growthList.appendChild(listItem);

                }

            }

           

            function updateLeaderboard() {

                // Get existing leaderboard from localStorage

                let leaderboard = JSON.parse(localStorage.getItem('dxcLeaderboard')) || [];

               

                // Add current player

                leaderboard.push({

                    name: gameState.playerName,

                    avatar: gameState.avatar,

                    score: gameState.score,

                    date: new Date().toISOString()

                });

               

                // Sort by score (highest first)

                leaderboard.sort((a, b) => b.score - a.score);

               

                // Keep top 5

                leaderboard = leaderboard.slice(0, 5);

               

                // Save back to localStorage

                localStorage.setItem('dxcLeaderboard', JSON.stringify(leaderboard));

               

                // Display leaderboard

                leaderboardContainer.innerHTML = '';

               

                if (leaderboard.length === 0) {

                    const message = document.createElement('p');

                    message.className = 'text-gray-500 italic';

                    message.textContent = "No results yet. Be the first on the leaderboard!";

                    leaderboardContainer.appendChild(message);

                } else {

                    leaderboard.forEach((entry, index) => {

                        const entryElement = document.createElement('div');

                        entryElement.className = 'leaderboard-entry flex items-center';

                       

                        // Position

                        const position = document.createElement('div');

                        position.className = 'font-bold text-2xl mr-4 dxc-text';

                        position.textContent = index + 1;

                       

                        // Avatar

                        const avatar = document.createElement('div');

                        avatar.className = 'w-10 h-10 rounded-full overflow-hidden border-2 border-gray-200 mr-4';

                       

                        const avatarImg = document.createElement('img');

                        if (entry.avatar === 'male') {

                            avatarImg.src = 'https://cdn.jsdelivr.net/gh/twitter/twemoji@14.0.2/assets/72x72/1f468-1f4bb.png';

                        } else if (entry.avatar === 'female') {

                            avatarImg.src = 'https://cdn.jsdelivr.net/gh/twitter/twemoji@14.0.2/assets/72x72/1f469-1f4bb.png';

                        } else {

                            avatarImg.src = 'https://cdn.jsdelivr.net/gh/twitter/twemoji@14.0.2/assets/72x72/1f9d1-1f4bb.png';

                        }

                        avatarImg.className = 'w-full h-full p-1';

                        avatar.appendChild(avatarImg);

                       

                        // Info

                        const info = document.createElement('div');

                        info.className = 'flex-grow';

                       

                        const name = document.createElement('div');

                        name.className = 'font-medium';

                        name.textContent = entry.name;

                       

                        const date = document.createElement('div');

                        date.className = 'text-sm text-gray-500';

                        date.textContent = new Date(entry.date).toLocaleDateString();

                       

                        info.appendChild(name);

                        info.appendChild(date);

                       

                        // Score

                        const score = document.createElement('div');

                        score.className = 'coin ml-auto';

                       

                        const coinIcon = document.createElement('i');

                        coinIcon.className = 'fas fa-coins coin-icon';

                       

                        const scoreText = document.createElement('span');

                        scoreText.textContent = entry.score;

                       

                        score.appendChild(coinIcon);

                        score.appendChild(document.createTextNode(' '));

                        score.appendChild(scoreText);

                       

                        // Add all elements to entry

                        entryElement.appendChild(position);

                        entryElement.appendChild(avatar);

                        entryElement.appendChild(info);

                        entryElement.appendChild(score);

                       

                        // Highlight current player

                        if (entry.name === gameState.playerName &&

                            entry.score === gameState.score &&

                            entry.avatar === gameState.avatar) {

                            entryElement.classList.add('bg-purple-50', 'border-l-4', 'border-purple-300');

                        }

                       

                        leaderboardContainer.appendChild(entryElement);

                    });

                }

            }

            }

           

            function createAnimatedAvatar() {

                animatedAvatarContainer.innerHTML = '';

               

                const avatar = document.createElement('div');

                avatar.className = 'animated-avatar floating';

               

                const avatarImg = document.createElement('img');

                if (gameState.avatar === 'male') {

                    avatarImg.src = 'https://cdn.jsdelivr.net/gh/twitter/twemoji@14.0.2/assets/72x72/1f468-1f4bb.png';

                } else if (gameState.avatar === 'female') {

                    avatarImg.src = 'https://cdn.jsdelivr.net/gh/twitter/twemoji@14.0.2/assets/72x72/1f469-1f4bb.png';

                } else {

                    avatarImg.src = 'https://cdn.jsdelivr.net/gh/twitter/twemoji@14.0.2/assets/72x72/1f9d1-1f4bb.png';

                }

                avatarImg.className = 'w-full h-full';

               

                avatar.appendChild(avatarImg);

                animatedAvatarContainer.appendChild(avatar);

            }

           

            function animateAvatar() {

                const maxWidth = gameScreen.offsetWidth - 150;

                const maxHeight = gameScreen.offsetHeight - 150;

               

                // Generate new random position

                gameState.avatarPosition = {

                    x: Math.random() * maxWidth,

                    y: Math.min(100, Math.random() * maxHeight / 2)  // Keep avatar in upper part of the screen

                };

               

                const avatar = document.querySelector('.animated-avatar');

                if (avatar) {

                    avatar.style.left = `${gameState.avatarPosition.x}px`;

                    avatar.style.top = `${gameState.avatarPosition.y}px`;

                }

            }

           

            function showHint() {

                if (gameState.currentScenario < scenarios.length && !feedbackContainer.classList.contains('show')) {

                    hintContainer.classList.remove('hidden');

                    hintContainer.classList.add('fade-in');

                   

                    // Hide hint after 5 seconds

                    setTimeout(() => {

                        hintContainer.classList.add('hidden');

                    }, 5000);

                }

            }

           

            function createConfetti(count) {

                confettiContainer.innerHTML = '';

               

                for (let i = 0; i < count; i++) {

                    const confetti = document.createElement('div');

                    confetti.className = 'confetti';

                   

                    // Random properties

                    const size = Math.random() * 10 + 5;

                    const colorIndex = Math.floor(Math.random() * gameState.confettiColors.length);

                    const left = Math.random() * 100;

                    const delay = Math.random() * 3;

                   

                    // Apply styles

                    confetti.style.width = `${size}px`;

                    confetti.style.height = `${size}px`;

                    confetti.style.backgroundColor = gameState.confettiColors[colorIndex];

                    confetti.style.left = `${left}%`;

                    confetti.style.animationDelay = `${delay}s`;

                   

                    confettiContainer.appendChild(confetti);

                }

            }

           

            function updateScore() {

                scoreDisplay.textContent = gameState.score;

            }

           

            function toggleMusic() {

                if (gameState.musicPlaying) {

                    backgroundMusic.pause();

                    musicToggle.innerHTML = '<i class="fas fa-volume-mute"></i>';

                    gameState.musicPlaying = false;

                } else {

                    backgroundMusic.play();

                    musicToggle.innerHTML = '<i class="fas fa-volume-up"></i>';

                    gameState.musicPlaying = true;

                }

            }

           

            function playSound(sound) {

                if (gameState.musicPlaying) {

                    sound.currentTime = 0;

                    sound.play();

                }

            }

           

            function shakeThing(element) {
